   * This is buzhash the hash function on which most other Hash methods are
   * built.
   */
  private static long buzhash(StringBuffer arg) {
    /* Hash StringBuffer */
    long h = initial_hash;
    for (int i = 0; i < arg.length(); ++i)
      h = (h << 1) ^ (h >>> 63) ^ mix_master[(arg.charAt(i) ^ (arg.charAt(i) >>> 8)) & 0xff];
    return h;
  }
